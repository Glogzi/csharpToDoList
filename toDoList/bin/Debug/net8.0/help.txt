help - "shows all commands"
add [what you want to add] - "adds to an to do list"
mark [line ID] - "marks thing you just did"
unmark [line ID] - "unmarks marked thing"
del [line ID] - "deletes something from your list"
edit [line ID] [edit to what] - "edits line (edited line will lose it's mark)"
delAll - "deletes all to Do list"
show - "shows all of the to Do list"
